Task 5

  1. Write a function for data summary
summary_bp<-function(data_list,stats='mean'){
  
  my_fun <- get(stats)
  
  treatment_pre  <- my_fun(data_list$treatment$pre_bp)
  treatment_post <- my_fun(data_list$treatment$post_bp)
  treatment_diff <- my_fun(data_list$treatment$diff_bp)
  placebo_pre  <- my_fun(data_list$placebo$pre_bp)
  placebo_post <- my_fun(data_list$placebo$post_bp)
  placebo_diff <- my_fun(data_list$placebo$diff_bp)
  
  out<-c(treatment_pre,treatment_post,treatment_diff,
         placebo_pre,placebo_post,placebo_diff)
  
  names(out)<-c(paste('pre treatment blood pressure',stats),
                paste('post treatment blood pressure' ,stats),
                paste('difference between treatment blood pressure',stats) ,
                paste('pre placebo blood pressure',stats),
                paste('post placebo blood pressure',stats) ,
                paste('difference between placebo blood pressure',stats ))
  return(out)
}
